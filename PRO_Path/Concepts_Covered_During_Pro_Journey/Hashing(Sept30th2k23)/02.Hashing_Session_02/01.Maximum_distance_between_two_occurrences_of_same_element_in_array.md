Brute Force

```cpp

int Sol(int &vec,int size)
{

	int max_dist=0;

	for(int i=0;i<size;++i)
	{
	  for(int j=i+1;j<size;++j)
		{
		 if(vec[i]==vec[j])
			{
			  max_dist = abs(i-j)>max_dist ? abs(i-j):max_dist;
			}
		}
	}
	return max_dist;
}
```

Optimized Solution

```cpp

int sol(int &vec, int size)
{
  unordered_map<int,int> mp;

  int max_dist=0;

  for(int i=0;i<size;++i)
  {
	if(mp.find(vec[i])==mp.end())
	{
	 mp[vec[i]]=i;
	}
	else
	{
	 max_dist=max(max_dist,i-mp(vec[i]));
	}
  }
	return max_dist;
}
```
